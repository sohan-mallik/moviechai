        function processFiles(files) {

            Array.from(files).forEach(file => {

                if (file.type.startsWith('video/')) {

                    const movie = {

                        id: Date.now() + Math.random(),

                        title: file.name.replace(/\.[^/.]+$/, ""),

                        year: new Date().getFullYear().toString(),

                        duration: "Unknown",

                        rating: "HD",

                        description: "User uploaded content",

                        url: URL.createObjectURL(file),

                        category: "upload",

                        file: file

                    };

                   

                    movies.push(movie);

                    populateRows();

                }

            });

           

            document.getElementById('uploadSection').style.display = 'none';

            document.getElementById('fileInput').value = '';

        }

        function openModal(movie) {

            currentMovie = movie;

            document.getElementById('modalTitle').textContent = movie.title;

            document.getElementById('modalRating').textContent = movie.rating || 'HD';

            document.getElementById('modalDuration').textContent = movie.duration || 'Unknown';

            document.getElementById('modalDescription').textContent = movie.description || 'No description available.';

           

            if (movie.url) {

                document.getElementById('modalVideo').src = movie.url;

            }

           

            document.getElementById('movieModal').style.display = 'block';

            document.body.style.overflow = 'hidden';

        }
